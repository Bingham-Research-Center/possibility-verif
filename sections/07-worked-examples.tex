\subsection{Scenario Setup}\label{sec:scenario_setup}

Three stylised possibilistic convective outlooks are defined, each
representing a distinct forecasting situation.  All distributions are
defined over $\Omega = \{\spc{MRGL},\, \spc{SLGT},\, \spc{ENH},\,
\spc{MDT},\, \spc{HIGH}\}$ with $K = 5$.

\paragraph{Scenario A: Sharp-correct \spc{MDT} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a textbook severe
%  weather day with strong instability, deep-layer shear, and a
%  well-defined mesoscale forcing mechanism.  The system produces a
%  confident, sharp forecast centered on \spc{MDT}.]

\[
\pi_A = \bigl\{\spc{MRGL}: 0.0,\; \spc{SLGT}: 0.05,\;
    \spc{ENH}: 0.15,\; \spc{MDT}: 0.90,\; \spc{HIGH}: 0.10\bigr\}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{MDT}$.

\paragraph{Scenario B: Hedged-correct \spc{ENH} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a messy environment
%  with competing signals: moderate instability but uncertain mesoscale
%  trigger timing.  The system hedges across multiple categories.]

\[
\pi_B = \bigl\{\spc{MRGL}: 0.10,\; \spc{SLGT}: 0.40,\;
    \spc{ENH}: 0.55,\; \spc{MDT}: 0.30,\; \spc{HIGH}: 0.0\bigr\}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{ENH}$.

\paragraph{Scenario C: Sharp-wrong \spc{MRGL} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a seemingly benign
%  environment that produces an unexpected severe outbreak.  The system
%  confidently predicts \spc{MRGL} but \spc{MDT} verifies.]

\[
\pi_C = \bigl\{\spc{MRGL}: 0.85,\; \spc{SLGT}: 0.10,\;
    \spc{ENH}: 0.0,\; \spc{MDT}: 0.0,\; \spc{HIGH}: 0.0\bigr\}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{MDT}$.
