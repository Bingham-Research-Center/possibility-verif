\subsection{Scenario Setup}\label{sec:scenario_setup}

Three stylised possibilistic convective outlooks are defined, each
representing a distinct forecasting situation.  All distributions are
defined over $\Omega = \{\spc{MRGL},\, \spc{SLGT},\, \spc{ENH},\,
\spc{MDT},\, \spc{HIGH}\}$ with $K = 5$.

\paragraph{Scenario A: Sharp-correct \spc{MDT} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a textbook severe
%  weather day with strong instability, deep-layer shear, and a
%  well-defined mesoscale forcing mechanism.  The system produces a
%  confident, sharp forecast centered on \spc{MDT}.]

\[
\pi_A(\omega) = \begin{cases}
0.00 & \omega = \spc{MRGL} \\
0.05 & \omega = \spc{SLGT} \\
0.15 & \omega = \spc{ENH}  \\
0.90 & \omega = \spc{MDT}  \\
0.10 & \omega = \spc{HIGH}
\end{cases}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{MDT}$.

\paragraph{Scenario B: Hedged-correct \spc{ENH} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a messy environment
%  with competing signals: moderate instability but uncertain mesoscale
%  trigger timing.  The system hedges across multiple categories.]

\[
\pi_B(\omega) = \begin{cases}
0.10 & \omega = \spc{MRGL} \\
0.40 & \omega = \spc{SLGT} \\
0.55 & \omega = \spc{ENH}  \\
0.30 & \omega = \spc{MDT}  \\
0.00 & \omega = \spc{HIGH}
\end{cases}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{ENH}$.

\paragraph{Scenario C: Sharp-wrong \spc{MRGL} forecast.}

% [PLACEHOLDER: Describe the meteorological setup — a seemingly benign
%  environment that produces an unexpected severe outbreak.  The system
%  confidently predicts \spc{MRGL} but \spc{MDT} verifies.]

\[
\pi_C(\omega) = \begin{cases}
0.85 & \omega = \spc{MRGL} \\
0.10 & \omega = \spc{SLGT} \\
0.00 & \omega = \spc{ENH}  \\
0.00 & \omega = \spc{MDT}  \\
0.00 & \omega = \spc{HIGH}
\end{cases}
\]
Observed outcome: $c_{\mathrm{obs}} = \spc{MDT}$.

\subsection{Filling the Gauge}\label{sec:filling_gauge}

% [PLACEHOLDER: 1–2 paragraphs introducing the gauge visualisation — each
%  SPC category is a horizontal bar whose fill level represents possibility.
%  The gap between the tallest bar and 1.0 is ignorance.  The observed
%  category is highlighted.]

Figure~\ref{fig:filling_gauge} shows the gauge visualisation for
Scenario~A.  The horizontal bars are filled to the raw possibility
level; the gap from $\Pi_{\max}$ to $1.0$ visually encodes the
ignorance $\ign$.  The observed category (\spc{MDT}) is highlighted
in green.

\begin{figure}[t]
\centering
\includegraphics[width=\columnwidth]{fig3_filling_gauge.pdf}
\caption{Gauge visualisation for Scenario~A (sharp-correct \spc{MDT}).
    Each horizontal bar represents an SPC category, filled to its raw
    possibility value $\pi(\omega)$.  The vertical dashed line marks
    $\Pi_{\max} = 0.90$; the gap to $1.0$ is ignorance $\ign = 0.10$.
    The observed category (\spc{MDT}, green) dominates the distribution,
    confirming high conditional necessity.}
\label{fig:filling_gauge}
\end{figure}
